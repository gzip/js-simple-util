179                 }
180                 obj = obj[prop];
181             }
182             
183             return obj;

191          * @return {object}

205                         } else {
206                             mergeTo[prop] = mergeFrom[prop];
207                         }
208                     }
209                 }
210             }
211             
212             return mergeTo;
213         },

505         /**
506          * Set attributes and properties on an element.
507          * @param {object} el DOM Node
508          * @param {object} attrs Key/value pairs including special handling for className,
509          *  innerHTML, parentNode, and styles. Anything else will be set as an attribute.
510          * @param {object} [events] Optional collection of listener functions keyed by event name.
511          * @return {object} The DOM Node.
512          */
513         setAttrs : function(el, attrs, events)
514         {
515             if (el) {
516                 if (attrs) {
517                     var attr, attribute;
518                     for (attr in attrs) {
519                         attribute = attrs[attr];
520                         switch(attr)
521                         {
522                             case cln:
523                             case innerHtml:
524                                 el[attr] = attribute;
525                             break;
526                             case 'parentNode':
527                                 attribute && attribute.appendChild(el);
528                             break;
529                             case 'styles':
530                                 util.setStyles(el, attribute, true);
531                             break;
532                             default:
533                                 el.setAttribute(attr, attribute);
534                             break;
535                         }
536                     }
537                 }
538                 if (events) {
539                     for (var ev in events) {
540                         util.listen(el, ev, events[ev]);
541                     }
542                 }
543             }
544             return el;
545         },

602         /**
603          * Make an XHR request.
604          * @param  {string} url URL.
605          * @param  {object} [opts] Options containing any of: callback (function),
606          *  data (string), headers (object), json (object), method (string),
607          *  parseHeaders (boolean), and props (object).
608          * @return {object} XMLHttpRequest object.
609          */
610         request : function (url, opts)
611         {
612             opts = opts || {};
613             var req = new XMLHttpRequest(),
614                 cb = opts.callback,
615                 method = opts.method || 'GET',
616                 headers = opts.headers || {},
617                 data = opts.data || null,
618                 json = opts.json;
619             
620             if (!util.isFunc(cb)) {
621                 return req;
622             }
623             
624             if (!util.isUnd(json)) {
625                 headers['Content-Type'] = 'application/json';
626                 try {
627                     data = JSON.stringify(json);
628                 } catch(e) {
629                     if (cb) {
630                         cb(e, null, req);
631                     }
632                 }
633             }
634             
635             if (url)
636             {
637                 req.open(method, url);
638                 
639                 util.each(headers, function(value, key)
640                 {
641                     req.setRequestHeader(key, value);
642                 });
643                 util.each(opts.props, function(value, key)
644                 {
645                     req[key] = value;
646                 });
647                 
648                 req.onreadystatechange = function()
649                 {
650                     var resp, status, headers;
651                     if (req.readyState === 4) {
652                         status = req.status;
653                         // TODO follow redirects? (opts.follow, opts.depth)
654                         if (status < 300 & status > 199) {
655                             if (opts.parseHeaders) {
656                                 req.headers = {};
657                                 headers = req.getAllResponseHeaders().split("\n");
658                                 util.each(headers, function(hdr)
659                                 {
660                                     var parsed = hdr.split(': ');
661                                     if (parsed.length === 2) {
662                                         req.headers[parsed[0].toLowerCase()] = parsed[1];
663                                     }
664                                 });
665                             }
666                             
667                             resp = req.responseText;
668                             
669                             if (!util.isUnd(json)) {
670                                 try {
671                                     resp = JSON.parse(resp);
672                                 } catch(e) {
673                                     cb(e, resp, req);
674                                 }
675                             }
676                             
677                             cb(null, resp, req);
678                         } else {
679                             cb({status: status, message: 'Non-200 returned.'}, null, req);
680                         }
681                     }
682                 };
683                 req.send(data);
684             }
685             
686             return req;
687         },

689         /**
690          * Listen to an event.
691          * @param  {object}   obj DOM Node to listen for events on.
692          * @param  {string}   type Event type, e.g. "click".
693          * @param  {function} fn Callback function.
694          */
695         listen : function(obj, type, fn)
696         {
697             if(!util.isObj(obj)) return;
698             
699             var add = 'addEventListener',
700                 attach = 'attachEvent',
701                 ev = 'on'+type;
702             
703             if (obj[add]) {
704                 obj[add](type, fn, false);
705             } else if (obj[attach]) {
706                 obj[attach](ev, function() { fn(window.event); } );
707             } else {
708                 obj[ev] = fn;
709             }

713          * Process an event.
714          * @param  {object} e Event object.
715          * @param  {boolean} [prevent] Prevent default.
716          * @param  {boolean} [stop] Stop propagation.

720         {
721             e = e || window.event;

723             if (prevent) {
724                 if (e.preventDefault) {

728                     }
729                 }

731                 if (stop) {
732                     e.cancelBubble = true;
733                 }
734             }
735             return el;

770         /**
771          * Get the  offset position of an element.
772          * @param  {object} el DOM Node to get offset for.
773          * @return {object} Object containing "x" and "y".
774          */
775         getOffset : function(el)
776         {
777             var parentNode = el,
778                 offsetLeft = 'offsetLeft',
779                 offsetTop = 'offsetTop',
780                 x = 0,
781                 y = 0;
782             
783             if (el) {
784                 x = el[offsetLeft];
785                 y = el[offsetTop];
786                 while(parentNode = parentNode.offsetParent) {
787                     x += parentNode[offsetLeft];
