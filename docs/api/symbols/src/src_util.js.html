142         /**
143          * Get a value [deep] in an object.
144          * @param  {object} obj Object to get a value from.
145          * @param  {string|array} path Property path, e.g. "foo.bar.baz".
146          * @param  {mixed} [def] Default value if nothing is found.
147          * @return {mixed} Value or default.

149         get : function(obj, path, def)
150         {
151             var isObj = util.isObj(obj),
152                 props = resolvePath(path),
153                 pl = isObj ? props.length : 0,
154                 p = 0;
155             
156             for (; p<pl && obj; p++)

159             }

165          * Set a value [deep] in an object.
166          * @param  {object} obj Object to set a value on.
167          * @param  {string|array} path Property path, e.g. "foo.bar.baz"
168          * @param  {mixed} val Value to set.
169          * @param {boolean} [conditional] Only set if not already present.
170          * @return {object} Object.
171          */
172         set : function(obj, path, val, conditional)
173         {
174             var props = util.isObj(obj) ? resolvePath(path) : [],
175                 prop,
176                 pl = props.length,
177                 p = 0;
178             
179             for (; p<pl; p++)
180             {

182                 if (!obj[owns](prop) || (p === pl-1 && !conditional)) {
183                     obj[prop] = p < pl-1 || util.isUnd(val) ? {} : val;
184                 }
185                 obj = obj[prop];
186             }
187             
188             return obj;

215         /**
216          * Merge properties from one object to another.
217          * @param  {object} mergeTo Object to merge to.
218          * @param  {object} mergeFrom Object to merge from.
219          * @param  {boolean} [clone] Clone the object rather than augment.
220          * @return {object}
221          */
222         merge : function(mergeTo, mergeFrom, clone)
223         {
224             var obj = util.isObj;
225             if (clone) {
226                 mergeTo = util.merge({}, mergeTo);
227             }
228             
229             if (obj(mergeTo) && obj(mergeFrom)) {
230                 for (var prop in mergeFrom) {
231                     if (mergeFrom[owns](prop)) {
232                         if (obj(mergeFrom[prop]) && !util.isNull(mergeTo[prop])) {
233                             mergeTo[prop] = util.merge(mergeTo[prop], mergeFrom[prop], clone);
234                         } else {
235                             mergeTo[prop] = mergeFrom[prop];
236                         }
237                     }
238                 }
239             }
240             
241             return mergeTo;
242         },

448

580         /**

590             if (el) {

592                     var attr, attribute;
593                     for (attr in attrs) {
594                         attribute = attrs[attr];
595                         switch(attr)
596                         {

607                             default:
608                                 el.setAttribute(attr, attribute);

610                         }
611                     }
612                 }
613                 if (events) {
614                     for (var ev in events) {
615                         util.listen(el, ev, events[ev]);
616                     }
617                 }
618             }

686         /**
687          * Make an XHR request.
688          * @param  {string} url URL.
689          * @param  {object} [opts] Options containing any of: callback (function),
690          *  data (string), headers (object), json (object), method (string),
691          *  parseHeaders (boolean), and props (object).
692          * @return {object} XMLHttpRequest object.
693          */
694         request : function (url, opts)
695         {
696             opts = opts || {};
697             var req = new XMLHttpRequest(),
698                 cb = opts.callback,
699                 method = opts.method || 'GET',
700                 headers = opts.headers || {},
701                 data = opts.data || null,
702                 json = opts.json;
703             
704             if (!util.isFunc(cb)) {
705                 return req;
706             }
707             
708             if (!util.isUnd(json)) {
709                 headers['Content-Type'] = 'application/json';
710                 try {
711                     data = JSON.stringify(json);
712                 } catch(e) {
713                     if (cb) {
714                         cb(e, null, req);
715                     }
716                 }
717             }
718             
719             if (url)
720             {
721                 req.open(method, url);
722                 
723                 util.each(headers, function(value, key)
724                 {
725                     req.setRequestHeader(key, value);
726                 });
727                 util.each(opts.props, function(value, key)
728                 {
729                     req[key] = value;
730                 });
731                 
732                 req.onreadystatechange = function()
733                 {
734                     var resp, status, headers;
735                     if (req.readyState === 4) {
736                         status = req.status;
737                         // TODO follow redirects? (opts.follow, opts.depth)
738                         if (status < 300 & status > 199) {
739                             if (opts.parseHeaders) {
740                                 req.headers = {};
741                                 headers = req.getAllResponseHeaders().split("\n");
742                                 util.each(headers, function(hdr)
743                                 {
744                                     var parsed = hdr.split(': ');
745                                     if (parsed.length === 2) {
746                                         req.headers[parsed[0].toLowerCase()] = parsed[1];
747                                     }
748                                 });
749                             }
750                             
751                             resp = req.responseText;
752                             
753                             if (!util.isUnd(json)) {
754                                 try {
755                                     resp = JSON.parse(resp);
756                                 } catch(e) {
757                                     cb(e, resp, req);
758                                 }
759                             }
760                             
761                             cb(null, resp, req);
762                         } else {
763                             cb({status: status, message: 'Non-200 returned.'}, null, req);
764                         }
765                     }
766                 };
767                 req.send(data);
768             }
769             
770             return req;
771         },

773         /**
774          * Listen to an event.
775          * @param  {object}   obj DOM Node to listen for events on.
776          * @param  {string}   type Event type, e.g. "click".
777          * @param  {function} fn Callback function.
778          */
779         listen : function(obj, type, fn)
780         {
781             if(!util.isObj(obj)) return;
782             
783             var add = 'addEventListener',
784                 attach = 'attachEvent',
785                 ev = 'on'+type;
786             
787             if (obj[add]) {
788                 obj[add](type, fn, false);

803         processEvent : function(e, prevent, stop)
804         {
805             e = e || window.event;
806             var el = e.target || e.srcElement;
807             if (prevent) {

813                 }
814                 e.returnValue = false;

818             }
